55// ConfObjectRef, needed by std_to_attr<T> when T is derived from ConnectBase
56template <>
57struct attr_from_std_helper<ConfObjectRef> {
58 static attr_value_t f(const ConfObjectRef &src) {
59 if (src) {
60 if (src.port_name().empty()) {
61 return SIM_make_attr_object(src);
62 } else {
63 attr_value_t ret = SIM_alloc_attr_list(2);
64 SIM_attr_list_set_item(&ret, 0, SIM_make_attr_object(src));
65 attr_value_t string = SIM_make_attr_string(
66 src.port_name().c_str());
67 SIM_attr_list_set_item(&ret, 1, string);
68 return ret;
69 }
70 } else {
71 return SIM_make_attr_nil();
72 }
73 }
74};
75
76template <>
77struct attr_to_std_helper<ConfObjectRef> {
78 static ConfObjectRef f(const attr_value_t &src) {
79 if (SIM_attr_is_list(src)) {
80 if (SIM_attr_list_size(src) != 2) {
81 throw SetIllegalType {
82 "Expected Simics list type with exactly 2 members"
83 };
84 }
85
86 auto item = SIM_attr_list_item(src, 0);
87 if (!SIM_attr_is_object(item)) {
88 throw SetIllegalType {
89 "The first item should be Simics object type"
90 };
91 }
92 ConfObjectRef r = SIM_attr_object(item);
93
94 item = SIM_attr_list_item(src, 1);
95 if (!SIM_attr_is_string(item)) {
96 throw SetIllegalType {
97 "The second item should be Simics string type"
98 };
99 }
100 r.set_port_name(SIM_attr_string(item));
101 return r;
102 } else {
103 if (!SIM_attr_is_object(src) && !SIM_attr_is_nil(src)) {
104 throw SetIllegalType {
105 "Expected Simics object or NIL type"
106 };
107 }
108 return SIM_attr_object_or_nil(src);
109 }
110 }

436// Container
437template <typename C, typename T = typename C::value_type>
438attr_value_t from_container(const C &src) {
439 auto it = src.cbegin();
440 const size_t size = src.size();
441 if (size > (std::numeric_limits<unsigned>::max)()) {
442 throw SetIllegalValue {
443 "Data too large to put in Simics list"
444 };
445 }
446 attr_value_t dst = SIM_alloc_attr_list(
447 static_cast<unsigned>(size));
448 for (unsigned i = 0; i < size; ++i, ++it) {
449 SIM_attr_list_set_item(&dst, i, std_to_attr(*it));
450 }
451 return dst;
452}

531template <typename... Args>
532struct attr_from_std_helper<std::tuple<Args...>> {
533 static attr_value_t f(const std::tuple<Args...> &src) {
534 attr_value_t dst = SIM_alloc_attr_list(sizeof...(Args));
535 set_items(&dst, src, 0, std::index_sequence_for<Args...>());
536 return dst;
537 }
538
539 private:
540 // Recursive template function to set items in the attribute list
541 template<std::size_t Index, std::size_t... Is>
542 static void set_items(attr_value_t *dst, const std::tuple<Args...> &src,
543 std::size_t current_index,
544 std::index_sequence<Index, Is...>) {
545 SIM_attr_list_set_item(dst, Index, std_to_attr(std::get<Index>(src)));
546 set_items(dst, src, current_index + 1, std::index_sequence<Is...>());
547 }
548
549 // Base case for recursion
550 static void set_items(attr_value_t *, const std::tuple<Args...> &,
551 std::size_t, std::index_sequence<>) {}
552};

554template <typename C, typename T = typename C::value_type>
555C to_container(const attr_value_t &src) {
556 if (!SIM_attr_is_list(src)) {
557 throw SetIllegalType {
558 "Expected Simics list type"
559 };
560 }
561 const unsigned size = SIM_attr_list_size(src);
562 std::vector<T> v;
563 for (unsigned i = 0; i < size; ++i) {
564 v.push_back(attr_to_std<T>(SIM_attr_list_item(src, i)));
565 }
566 return {v.begin(), v.end()};
567}

649template <typename... Args>
650struct attr_to_std_helper<std::tuple<Args...>> {
651 static std::tuple<Args...> f(const attr_value_t &src) {
652 if (!SIM_attr_is_list(src)
653 || SIM_attr_list_size(src) != sizeof...(Args)) {
654 throw SetIllegalType{
655 "Expected Simics list type with exactly " + \
656 std::to_string(sizeof...(Args)) + " members"
657 };
658 }
659
660 return convert_tuple(src, std::index_sequence_for<Args...>());
661 }
662
663 private:
664 // Helper function to convert items in the attribute list to a tuple
665 template<std::size_t... Is>
666 static std::tuple<Args...> convert_tuple(const attr_value_t &src,
667 std::index_sequence<Is...>) {
668 return std::make_tuple(convert_item<Args>(src, Is)...);
669 }
670
671 template<typename T>
672 static T convert_item(const attr_value_t &src, std::size_t index) {
673 return attr_to_std<T>(SIM_attr_list_item(src, index));
674 }
