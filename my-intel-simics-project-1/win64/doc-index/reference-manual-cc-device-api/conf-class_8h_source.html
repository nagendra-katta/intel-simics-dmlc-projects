204template <typename T> ConfClassPtr
205make_class(const std::string &name, const std::string &short_desc,
206 const std::string &description,
207 const class_kind_t kind = Sim_Class_Kind_Vanilla) {
208 auto cls = ConfClass::createInstance(name, short_desc, description, kind,
209 ObjectFactory<T>());
210 detail::init_class<T>(cls.get());
211 decorate_class<T>(nullptr, cls.get());
212 return cls;
213}

262 // This type is defined (as int) only when A is None
263 template <typename U = A>
264 using EnableIfNone = std::enable_if_t<std::is_same<U, None>::value, int>;
265
266 // This type is defined (as int) only when A is not None
267 template <typename U = A>
268 using EnableIfNotNone = std::enable_if_t<!std::is_same<U, None>::value,
269 int>;
270
271 public:
272 // Constructor for classes without additional arguments
273 template <typename U = A, EnableIfNone<U> = 0> constexpr
274 RegisterClassWithSimics(const std::string &name,
275 const std::string &short_desc,
276 const std::string &description,
277 const class_kind_t kind = Sim_Class_Kind_Vanilla) {
278 make_class<T>(name, short_desc, description, kind);

286 A *constructor_arg,
287 const class_kind_t kind = Sim_Class_Kind_Vanilla) {
