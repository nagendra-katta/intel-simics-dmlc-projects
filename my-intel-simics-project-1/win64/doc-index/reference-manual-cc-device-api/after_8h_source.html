32/*
33 * This class serves as a manager for a collection of interfaces that implement
34 * the `AfterCallInterface`. It provides static methods to add, remove,
35 * and find interface instances based on their names.
36 */
37class AfterCall {
38 public:
39 static void addIface(AfterCallInterface *iface) {
40 if (iface == nullptr) {
41 throw std::invalid_argument {
42 "AfterCallInterface pointer cannot be nullptr"
43 };
44 } else {
45 getIfaces().insert(iface);
46 }
47 }
48
49 static void removeIface(AfterCallInterface *iface) {
50 if (iface == nullptr) {
51 throw std::invalid_argument {
52 "AfterCallInterface pointer cannot be nullptr"
53 };
54 } else {
55 getIfaces().erase(iface);
56 }
57 }
58
59 static AfterCallInterface *findIface(const std::string &name) {
60 if (name.empty()) {
61 throw std::invalid_argument {
62 "Empty name cannot be used to find AfterCallInterface"
63 };
64 }
65 for (auto iface : getIfaces()) {
66 if (iface->name() == name) {
67 return iface;
68 }
69 }
70 return nullptr;
71 }
72
73 private:
74 // Use a function-local static pointer to the unordered_set per
75 // C++ style guideline since STL set has non-trivial destructors
76 static std::unordered_set<AfterCallInterface*> &getIfaces() {
77 static auto* ifaces = new std::unordered_set<
78 AfterCallInterface*>();
79 return *ifaces;
80 }
81};

83// FunctionCall contains information about a callable function, including
84// its name, arguments, and a pointer to the function itself.
85template<typename... Args>
86class FunctionCall : public AfterCallInterface {
87 public:
88 using FunctionType = void(*)(Args...);
89
90 FunctionCall(FunctionType func, const std::string &name)
91 : func_(func),
92 name_(name + typeid(func).name()) {}
93
94 FunctionCall(const FunctionCall &other)
95 : func_(other.func_), name_(other.name_), args_(other.args_) {}
96 FunctionCall& operator=(const FunctionCall& other) {
97 if (this != &other) {
98 // not a self-assignment
99 func_ = other.func_;
100 name_ = other.name_;
101 args_ = other.args_;
102 }
103 return *this;
104 }
105
106 std::string name() const override {
107 return name_;
108 }
109
110 void set_args(const attr_value_t &value) override {
111 args_ = attr_to_std<std::tuple<Args...>>(value);
112 }
113
114 AfterCallInterface *make_copy() override {
115 // The allocated pointer is deleted after the function call is invoked
116 return new FunctionCall<Args...>(*this);
117 }
118
119 attr_value_t get_value() override {
120 return std_to_attr(std::make_pair(name_, args_));
121 }
122
123 private:
124 // Invokes the stored function with the current arguments
125 void invoke() override {
126 invoke_impl(std::index_sequence_for<Args...>{});
127 }
128
129 // Helper function to unpack the tuple and call the function
130 template<std::size_t... I>
131 void invoke_impl(std::index_sequence<I...>) {
132 func_(std::get<I>(args_)...);
133 }
134
135 FunctionType func_;
136 std::string name_;
137 std::tuple<Args...> args_;
138};

140// MemberFunctionCall contains information about a class member function,
141// including a pointer to the object, the function's name, its arguments,
142// and a pointer to the member function itself
143template<typename Class, typename... Args>
144class MemberFunctionCall : public AfterCallInterface {
145 public:
146 using MemberFunctionType = void (Class::*)(Args...);
147
148 MemberFunctionCall(MemberFunctionType func, const std::string &name)
149 : func_(func), name_(name + typeid(func).name()),
150 obj_(nullptr) {}
151
152 MemberFunctionCall(const MemberFunctionCall& other)
153 : func_(other.func_), name_(other.name_), obj_(other.obj_),
154 args_(other.args_) {}
155 MemberFunctionCall& operator=(const MemberFunctionCall& other) {
156 if (this != &other) {
157 // not a self-assignment
158 func_ = other.func_;
159 name_ = other.name_;
160 obj_ = other.obj_;
161 args_ = other.args_;
162 }
163 return *this;
164 }
165
166 std::string name() const override {
167 return name_;
168 }
169
170 // Sets the arguments for the member function call from a given
171 // attribute value.
172 // The first element is treated as a reference to the object, and
173 // subsequent elements are treated as arguments for the member function
174 void set_args(const attr_value_t &value) override {
175 auto t = attr_to_std<std::tuple<ConfObjectRef, Args...>>(value);
176 obj_ = std::get<0>(t);
177 set_args_impl(t, std::index_sequence_for<Args...>{});
178 }
179
180 AfterCallInterface *make_copy() override {
181 // The allocated pointer is deleted after the function call is invoked
182 return new MemberFunctionCall<Class, Args...>(*this);
183 }
184
185 attr_value_t get_value() override {
186 auto rest = std::tuple_cat(std::make_tuple(obj_), args_);
187 return std_to_attr(std::make_pair(name_, rest));
188 }
189
190 private:
191 void invoke() override {
192 if (!obj_) {
193 throw std::invalid_argument {
194 "Cannot call class member function without class instance"
195 };
196 }
197 invoke_impl(std::index_sequence_for<Args...>{});
198 }
199
200 // Helper function to unpack the tuple and call the function
201 template<std::size_t... I>
202 void invoke_impl(std::index_sequence<I...>) {
203 (from_obj<Class>(obj_)->*func_)(std::get<I>(args_)...);
204 }
205
206 // Helper function to return a tuple by removing the first item
207 template <std::size_t... Is>
208 void set_args_impl(const std::tuple<ConfObjectRef, Args...> &input_tuple,
209 std::index_sequence<Is...>) {
210 args_ = std::make_tuple(std::get<Is + 1>(input_tuple)...);
211 }
212
213 MemberFunctionType func_;
214 std::string name_;
215 ConfObjectRef obj_;
216 std::tuple<Args...> args_;
217};

222class BankInterface;
223class RegisterInterface;
224class FieldInterface;
225template<typename Class, typename... Args>
226constexpr void check_function_call(void(Class::*func)(Args...)) {
227 static_assert(std::is_base_of<ConfObject, Class>::value
228 || std::is_base_of<BankInterface, Class>::value
229 || std::is_base_of<RegisterInterface, Class>::value
230 || std::is_base_of<FieldInterface, Class>::value,
231 "Only class derived of ConfObject/BankInterface/"
232 "RegisterInterface/FieldInterface supports the after call");
233}

284 // AfterInterface
285 void schedule(double seconds, const std::string &name,
286 const attr_value_t &args) override {
287 auto *iface = get_iface(name);
288 if (iface) {
289 iface->set_args(args);
290 after_event.post(seconds, iface);
291 }
292 }
293 void schedule(cycles_t cycles, const std::string &name,
294 const attr_value_t &args) override {
295 auto *iface = get_iface(name);
296 if (iface) {
297 iface->set_args(args);
298 after_event.post(cycles, iface);
299 }
300 }
301 void cancel_all() override {
302 after_event.remove();
303 }
304
305 static event_class_t *event_cls;
306
307 protected:
308 ConfObject *obj_ {nullptr};
309 AfterEvent after_event;
310
311 private:
312 // Get AfterCallInterface by name with the arguments configured
313 AfterCallInterface *get_iface(const std::string &name) {
314 auto *iface = AfterCall::findIface(name);
315 if (iface == nullptr) {
316 SIM_LOG_ERROR_STR(
317 obj_->obj(), 0,
318 std::string("After call (" + name + ") needs to be "
319 "registered by REGISTER_AFTER_CALL or "
320 "REGISTER_REG_BANK_AFTER_CALL first"));
321 return nullptr;
322 }
323 return iface->make_copy();
324 }
325};
