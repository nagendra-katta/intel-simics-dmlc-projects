35// RegBankFunctionCall contains class member function for bank/register/field
36// information like object name, function name, arguments and the function
37// pointer
38template<typename Class, typename... Args>
39class RegBankFunctionCall : public AfterCallInterface {
40 public:
41 using RegBankFunctionType = void (Class::*)(Args...);
42
43 RegBankFunctionCall(RegBankFunctionType func, const std::string &name)
44 : func_(func), name_(name + typeid(func).name()),
45 obj_(nullptr), dev_obj_(nullptr), hierarchical_object_name_("") {}
46
47 RegBankFunctionCall(const RegBankFunctionCall &other)
48 : func_(other.func_), name_(other.name_),
49 obj_(other.obj_), dev_obj_(other.dev_obj_),
50 hierarchical_object_name_(other.hierarchical_object_name_),
51 args_(other.args_) {}
52 RegBankFunctionCall& operator=(const RegBankFunctionCall& other) {
53 if (this != &other) {
54 // not a self-assignment
55 func_ = other.func_;
56 name_ = other.name_;
57 obj_ = other.obj_;
58 dev_obj_ = other.dev_obj_;
59 hierarchical_object_name_ = other.hierarchical_object_name_;
60 args_ = other.args_;
61 }
62 return *this;
63 }
64
65 std::string name() const override {
66 return name_;
67 }
68
69 void set_args(const attr_value_t &value) override {
70 auto t = attr_to_std<std::tuple<
71 ConfObjectRef, std::string, Args...>>(value);
72 dev_obj_ = std::get<0>(t);
73 hierarchical_object_name_ = std::get<1>(t);
74 set_args_impl(t, std::index_sequence_for<Args...>{});
75 }
76
77 AfterCallInterface *make_copy() override {
78 // The allocated pointer is deleted after the function call is invoked
79 return new RegBankFunctionCall<Class, Args...>(*this);
80 }
81
82 attr_value_t get_value() override {
83 auto rest = std::tuple_cat(
84 std::make_tuple(dev_obj_, hierarchical_object_name_),
85 args_);
86 return std_to_attr(std::make_pair(name_, rest));
87 }
88
89 private:
90 void invoke() override {
91 auto level = static_cast<Level>(
92 HierarchicalObject::level_of_hierarchical_name(
93 hierarchical_object_name_));
94 auto *dev = from_obj<MappableConfObject>(dev_obj_);
95 switch (level) {
96 case Level::BANK:
97 obj_ = dynamic_cast<Class *>(dev->get_iface<BankInterface>(
98 hierarchical_object_name_));
99 break;
100 case Level::REGISTER:
101 obj_ = dynamic_cast<Class *>(dev->get_iface<RegisterInterface>(
102 hierarchical_object_name_));
103 break;
104 case Level::FIELD:
105 obj_ = dynamic_cast<Class *>(dev->get_iface<FieldInterface>(
106 hierarchical_object_name_));
107 break;
108 default:
109 SIM_LOG_ERROR(dev_obj_, 0,
110 "%s is not a valid hierarchical object name",
111 hierarchical_object_name_.c_str());
112 }
113 invoke_impl(std::index_sequence_for<Args...>{});
114 }
115
116 // Helper function to unpack the tuple and call the function
117 template<std::size_t... I>
118 void invoke_impl(std::index_sequence<I...>) {
119 (obj_->*func_)(std::get<I>(args_)...);
120 }
121
122 // Helper function to return a tuple by removing the first item
123 template <std::size_t... Is>
124 void set_args_impl(const std::tuple<ConfObjectRef, std::string,
125 Args...>& input_tuple,
126 std::index_sequence<Is...>) {
127 args_ = std::make_tuple(std::get<Is + 2>(input_tuple)...);
128 }
129
130 RegBankFunctionType func_;
131 std::string name_;
132 Class *obj_ {nullptr};
133 ConfObjectRef dev_obj_;
134 std::string hierarchical_object_name_;
135 std::tuple<Args...> args_;
136};
137

139// Helper function to create a RegBankFunctionCall object with deduced types
140template<typename Class, typename... Args>
141constexpr auto make_reg_bank_function_call(void (Class::*func)(Args...),
142 const std::string &name) {
143 return new RegBankFunctionCall<Class, Args...>(func, name);
144}
