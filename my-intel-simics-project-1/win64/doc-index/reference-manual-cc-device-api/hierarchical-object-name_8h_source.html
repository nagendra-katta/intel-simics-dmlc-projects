73 static constexpr void validate_name(std::string_view name) {
74#ifdef __GNUC__
75 using std::isalpha;
76 using std::isalnum;
77#endif
78 if (name.empty()) {
79 throw std::invalid_argument("Empty name is not allowed");
80 }
81 if (!isalpha(name.front())) {
82 throw std::invalid_argument(
83 std::string("Name (") + name.data() \
84 + ") does not begin with an alphabetic character");
85 }
86
87 // Validate base_name
88 for (const auto &c : name.substr(0, name.find('['))) {
89 if (c != '_' && !isalnum(c)) {
90 throw std::invalid_argument(
91 std::string("Character (") + c \
92 + ") is not allowed to use in a name");
93 }
94 }
95 }

125 std::map<std::string, size_t> arrayNamesToOffsets(size_t width) const {
126 if (width == 0) {
127 throw std::invalid_argument("Invalid width 0");
128 }
129
130 if (array_str().empty()) {
131 return {};
132 }
133
134 auto sizes_and_strides = arraySizesAndStrides();
135 // Update strides for all dimensions
136 for (auto it = sizes_and_strides.rbegin();
137 it != sizes_and_strides.rend(); ++it) {
138 if (it->second != 0) {
139 continue;
140 }
141 if (it == sizes_and_strides.rbegin()) {
142 // The innermost dimension, use width as the stride
143 it->second = width;
144 } else {
145 // Use the sum of distances of inner dimension elements
146 it->second = (it - 1)->first * (it - 1)->second;
147 }
148 }
149
150 std::map<std::string, size_t> names_to_offsets;
151 std::vector<size_t> indices(sizes_and_strides.size(), 0);
152 generateNamesToOffsets(sizes_and_strides, 0, &indices,
153 &names_to_offsets);
154 return names_to_offsets;
155 }

171 std::vector<std::pair<size_t, size_t>> arraySizesAndStrides() const {
172 auto s = array_str();
173 if (s.empty()) {
174 return {};
175 }
176
177 std::size_t content_pos = npos;
178 std::vector<std::pair<size_t, size_t>> dims_;
179 for (size_t i = 0; i < s.size(); ++i) {
180 auto c = s[i];
181 if (c == '[') {
182 if (content_pos != npos) {
183 throw std::logic_error("Name has unbalanced brackets");
184 }
185 content_pos = i + 1;
186 } else if (c == ']') {
187 if (content_pos == npos) {
188 throw std::logic_error("Name has unbalanced brackets");
189 }
190 if (content_pos == i) {
191 throw std::logic_error("Name has nothing in brackets");
192 }
193 const auto &[size, stride] = sizeAndStride(
194 s.substr(content_pos, i - content_pos));
195 if (size == 0) {
196 throw std::logic_error("Dimension size is 0");
197 }
198 dims_.push_back({size, stride});
199 content_pos = npos;
200 }
201 }
202
203 if (content_pos != npos) {
204 throw std::logic_error("Name has unbalanced brackets");
205 }
206
207 return dims_;
208 }

236 std::pair<size_t, size_t> sizeAndStride(std::string_view s) const {
237 auto stride_pos = s.find(" stride ");
238 size_t size;
239 size_t stride = 0;
240
241 try {
242 if (stride_pos != npos) {
243 size = std::stoi(s.substr(0, stride_pos).data());
244 stride = std::stoi(s.substr(stride_pos + 8).data());
245 } else {
246 if (s.find_first_not_of("0123456789") != npos) {
247 throw std::invalid_argument("non-digit character");
248 }
249 size = std::stoi(s.data());
250 }
251 } catch (const std::exception &e) {
252 throw std::invalid_argument(
253 std::string("Array contents are malformed: ") + e.what());
254 }
255
256 return {size, stride};
257 }

283 void generateNamesToOffsets(const std::vector<std::pair<size_t, size_t>>
284 &dims_info,
285 size_t current_dim,
286 std::vector<size_t> *indices,
287 std::map<std::string, size_t>
288 *names_to_offsets) const {
289 if (current_dim == dims_info.size()) {
290 // We have reached the innermost dimension, so generate the name
291 // and offset and add it to the vector
292 std::string name {base_name()};
293 size_t offset = 0;
294 for (size_t i = 0; i < indices->size(); i++) {
295 name += "[" + std::to_string((*indices)[i]) + "]";
296 offset += (*indices)[i] * dims_info[i].second;
297 }
298 (*names_to_offsets)[name] = offset;
299 } else {
300 // Iterate over each index of the current dimension and recurse
301 for (size_t i = 0; i < dims_info[current_dim].first; i++) {
302 (*indices)[current_dim] = i;
303 generateNamesToOffsets(dims_info, current_dim + 1,
304 indices, names_to_offsets);
305 }
306 }
307 }
308};
