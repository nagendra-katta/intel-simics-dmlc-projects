15/*
16 * Used by device that simulates a PCIe endpoint/multifunciton endpoint with
17 * a Type 0 header.
18 * It implements the following Simics interfaces:
19 * - `pcie_device` interface which handles the PCIe connection
20 * - `pcie_transaction` interface which receives PCIe transactions
21 * - `signal` interface which receives warm reset
22 * It implements the following Simics connectors:
23 * - `pcie_map` connector which controls the map/unmap bar memory, and
24 * enable/disable PCI functions
25 * - `pcie_transaction` connector which sends PCIe transactions
26 */
27

57class PcieGasketBase {
58 public:
59 typedef PcieGasketBase *is_composite_pcie_gasket;
60 template <typename C>
61 static void initClassInternal(ConfClass *cls) {
62 static iface::PcieDeviceSimicsAdapter<C> pcie_device_simics_adapter;
63 static iface::TransactionSimicsAdapter<C> transaction_simics_adapter;
64 static iface::SignalSimicsAdapter<C> signal_simics_adapter;
65 cls->add(Attribute("upstream_target", "o|n",
66 "The PCIe upstream target to connect to.",
67 ATTR_CLS_VAR(C, simics_transaction_target_)));
68 cls->add(Attribute("address_id_mem_map", "[[[ii]i]*]",
69 "Internal. The map from MEM address range to"
70 " socket ID",
71 ATTR_GETTER(C, addressIdMemMap), nullptr));
72 cls->add(Attribute("address_id_io_map", "[[[ii]i]*]",
73 "Internal. The map from IO address range to"
74 " socket ID",
75 ATTR_GETTER(C, addressIdIoMap), nullptr));
76 cls->add(Attribute("enable_base_address_subtraction", "b",
77 "If set, MEM/IO transaction will receive only the"
78 " offset to the base address. Otherwise, it receives"
79 " the whole base address + offset. Default enabled",
80 ATTR_GETTER(C, enableBaseAddressSubtraction),
81 ATTR_SETTER(C, setEnableBaseAddressSubtraction)));
82 cls->add(pcie_device_simics_adapter);
83 cls->add(transaction_simics_adapter);
84 cls->add(signal_simics_adapter);
85
86 // Depends on the pcie_map_helper_cpp class
87 auto *map_helper = SIM_get_class("pcie_map_helper_cpp");
88 if (SIM_clear_exception() != SimExc_No_Exception) {
89 std::cerr << "ERROR: Simics class pcie_map_helper_cpp is not"
90 << " registered. Please build module"
91 << " pcie-map-helper-c++ first."
92 << std::endl;
93 }
94 assert(map_helper);
95 SIM_register_port(*cls, "port.mem", map_helper,
96 " The help object maps to the PCIe MEM space");
97 SIM_register_port(*cls, "port.io", map_helper,
98 "The help object maps to the PCIe IO space");
99 SIM_register_port(*cls, "port.msg", map_helper,
100 "The help object mapes to the PCIe MSG space");
101 }
102
103 protected:
104 Connector<tlm2simics::PcieTransaction> simics_transaction_target_;
105 ConnectorProxy<tlm2simics::PcieMap> simics_pcie_map_target_ {
106 &simics_transaction_target_
107 };

143 /*
144 * Intercepts PcieDeviceGasketAdapter to connect/disconnect the simics
145 * upstream target connector before pass it down to the IC
146 */
147 void connected(conf_object_t *port_obj, uint16_t id) override {
148 Context context(simulation_);
149
150 if (port_obj == nullptr) {
151 SIM_LOG_ERROR(simulation_->simics_object(), 0,
152 "can't connect to NULL");
153 return;
154 }
155
156 if (simics_transaction_target_) {
157 if (simics_transaction_target_ != port_obj) {
158 SIM_LOG_ERROR(
159 simulation_->simics_object(), 0,
160 "can't connect to '%s', currently connected to '%s'",
161 SIM_object_name(port_obj),
162 SIM_object_name(simics_transaction_target_));
163 return;
164 }
165 }
166
167 simics_transaction_target_.set(port_obj);
168 interconnect_.connected(id);
169 }
170
171 void disconnected(conf_object_t *port_obj, uint16_t id) override {
172 Context context(simulation_);
173
174 if (port_obj == nullptr) {
175 SIM_LOG_ERROR(simulation_->simics_object(), 0,
176 "can't disconnect from NULL");
177 return;
178 }
179
180 if (simics_transaction_target_ != port_obj) {
181 SIM_LOG_ERROR(simulation_->simics_object(), 0,
182 "can't disconnect from '%s', currently connected"
183 " to '%s'",
184 SIM_object_name(port_obj),
185 SIM_object_name(simics_transaction_target_));
186 return;
187 }
188
189 if (simics_transaction_target_) {
190 // Delete mapping
191 interconnect_.disconnected(id);
192 simics_transaction_target_.set(nullptr);
193 }
194 }
195
196 void hot_reset() override {
197 Context context(simulation_);
198 interconnect_.hotReset();
199 }

201 // NOTE: By using the TPciDevice as method parameter, the compiler can
202 // deduce the template parameter.
203 template<typename TPcieDevice>
204 void connect(TPcieDevice *device) {
205 auto o = simulation_->simics_object();
206
207 // Connect IC's pcie_device target socket
208 // No payload is sent to the socket since PcieDevice is completely
209 // handled here
210 systemc_pcie_.set_gasket(
211 simics2tlm::createGasket(
212 &interconnect_.pcie_device_target_socket, o));
213
214 // Connect IC's transaction target socket (snooping)
215 systemc_transaction_.set_gasket(
216 simics2tlm::createGasket(
217 &interconnect_.transaction_target_socket, o));
218
219 // Connect IC's warm reset target socket
220 systemc_signal_.set_pin(&interconnect_.warm_reset_pin, false, o);
221
222 // Connect IC's pcie_map initiator socket to the pcie upstream target
223 simics_transaction_target_->set_gasket(
224 tlm2simics::createMultiGasket(
225 &interconnect_.pcie_map_initiator_socket, o));
226 simics_pcie_map_target_->set_gasket(
227 tlm2simics::createMultiGasket(
228 &interconnect_.pcie_map_initiator_socket, o));
229
230 // Retrieve the required information from the device to configure
231 // the IC to prepare the connection with the device
232 interconnect_.connect(
233 dynamic_cast<iface::PcieDeviceQueryInterface *>(device),
234 dynamic_cast<iface::PcieBaseAddressRegisterQueryInterface *>(
235 device),
236 dynamic_cast<iface::PcieResetInterface *>(device),
237 obj_ ? obj_ : o);
238 }

261 SIM_make_attr_object(obj));
262 SIM_set_attribute_default(
263 SIM_object_descendant(obj, "port.io"), "pcie_type",
264 SIM_make_attr_int64(types::PCIE_Type_IO));
265 SIM_set_attribute_default(
266 SIM_object_descendant(obj, "port.io"), "forward_target",
267 SIM_make_attr_object(obj));
268 SIM_set_attribute_default(
269 SIM_object_descendant(obj, "port.msg"), "pcie_type",
270 SIM_make_attr_int64(types::PCIE_Type_Msg));
271 SIM_set_attribute_default(
272 SIM_object_descendant(obj, "port.msg"), "forward_target",
273 SIM_make_attr_object(obj));
